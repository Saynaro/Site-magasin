model Product {
  id    Int     @id @default(autoincrement())
  name  String
  price Float
}     
                    
                    BACKEND

const express = require('express');
const { PrismaClient } = require('@prisma/client');
const cors = require('cors');

const app = express();
const prisma = new PrismaClient();

app.use(cors());
app.use(express.json());

app.get('/products', async (req, res) => {
  try {
    let { sortBy = 'price', order = 'asc' } = req.query;

    const allowedFields = ['price', 'name'];
    const allowedOrder = ['asc', 'desc'];

    if (!allowedFields.includes(sortBy)) sortBy = 'price';
    if (!allowedOrder.includes(order)) order = 'asc';

    const products = await prisma.product.findMany({
      orderBy: {
        [sortBy]: order,
      },
    });

    res.json(products);
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Ошибка сервера' });
  }
});

app.listen(3000, () => {
  console.log('Server running on http://localhost:3000');
});



                                        FRONTEND
                            <!DOCTYPE html>
<html lang="ru">
<head>
  <meta charset="UTF-8">
  <title>Товары</title>
</head>
<body>

  <h1>Список товаров</h1>

  <select id="sort">
    <option value="">Без сортировки</option>
    <option value="price_asc">Цена ↑</option>
    <option value="price_desc">Цена ↓</option>
  </select>

  <div id="products"></div>

  <script>
    const select = document.getElementById('sort');
    const container = document.getElementById('products');

    select.addEventListener('change', () => {
      const value = select.value;

      let sortBy = 'price';
      let order = 'asc';

      if (value === 'price_asc') {
        sortBy = 'price';
        order = 'asc';
      } else if (value === 'price_desc') {
        sortBy = 'price';
        order = 'desc';
      }

      loadProducts(sortBy, order);
    });

    async function loadProducts(sortBy = 'price', order = 'asc') {
      const res = await fetch(
        `http://localhost:3000/products?sortBy=${sortBy}&order=${order}`
      );

      const products = await res.json();

      container.innerHTML = products.map(p => `
        <div style="border:1px solid #ccc; margin:10px; padding:10px;">
          <h3>${p.name}</h3>
          <p>Цена: ${p.price}</p>
        </div>
      `).join('');
    }

    // первая загрузка
    loadProducts();
  </script>

</body>
</html>


Что происходит
Фронт отправляет запрос:
/products?sortBy=price&order=asc
Бэкенд:
orderBy: { price: 'asc' }
Prisma сортирует данные в базе
Ты получаешь отсортированный список
